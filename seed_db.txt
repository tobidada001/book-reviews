tmp_dir = tempfile.mkdtemp()
ok = 0
err = 0

for title, author, genre, pub_date, isbn, pages, img_file in books_data:
    url = BASE_URL + img_file
    local_path = os.path.join(tmp_dir, img_file)
    try:
        urlretrieve(url, local_path)
        with open(local_path, 'rb') as f:
            book = Book(
                title=title,
                author=author,
                genre=genre,
                publication_date=pub_date,
                isbn=isbn,
                number_of_pages=pages,
            )
            book.cover_image.save(img_file, File(f), save=True)
        ok += 1
        print(f"OK  {title}")
    except Exception as e:
        err += 1
        print(f"ERR {title}: {e}")
        # Still create the book without the image
        Book.objects.create(
            title=title,
            author=author,
            genre=genre,
            publication_date=pub_date,
            isbn=isbn,
            number_of_pages=pages,
        )

shutil.rmtree(tmp_dir, ignore_errors=True)
print(f"\nDone! {Author.objects.count()} authors, {Book.objects.count()} books ({ok} with covers, {err} without).")
